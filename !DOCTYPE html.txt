<body>
    <h1>Google Sheets Data Filter</h1>
    <label for="filter">Filter by keyword:</label>
    <input type="text" id="filter" placeholder="Enter keyword">
    <button onclick="filterData()">Search</button>

    <h2>Results:</h2>
    <table id="resultsTable">
        <thead>
            <tr>
                <th>Name</th>
                <th>Age</th>
                <th>City</th>
            </tr>
        </thead>
        <tbody id="resultsBody">
            <!-- Filtered data will appear here -->
        </tbody>
    </table>

    <script>
        // Google Sheets ID and API key
        const SHEET_ID = 'your_google_sheet_id';
        const API_KEY = 'your_google_api_key';

        // URL to get data from Google Sheets
        const url = `https://sheets.googleapis.com/v4/spreadsheets/${SHEET_ID}/values/Sheet1?key=${API_KEY}`;

        let sheetData = [];

        // Fetch data from Google Sheets
        fetch(url)
            .then(response => response.json())
            .then(data => {
                sheetData = data.values;  // All data from the sheet
                displayData(sheetData);   // Display initial data
            })
            .catch(error => console.error('Error fetching data:', error));

        // Function to display data in the table
        function displayData(data) {
            const tableBody = document.getElementById('resultsBody');
            tableBody.innerHTML = ''; // Clear current table

            data.forEach(row => {
                const tr = document.createElement('tr');
                row.forEach(cell => {
                    const td = document.createElement('td');
                    td.textContent = cell;
                    tr.appendChild(td);
                });
                tableBody.appendChild(tr);
            });
        }

        // Filter function
        function filterData() {
            const keyword = document.getElementById('filter').value.toLowerCase();
            const filteredData = sheetData.filter(row => 
                row.some(cell => cell.toLowerCase().includes(keyword))
            );
            displayData(filteredData);
        }
    </script>
